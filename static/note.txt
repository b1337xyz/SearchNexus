pyinstaller build command [Windows]

  pyinstaller --windowed --clean --onefile --icon=static/favicon/favicon.ico --name SearchNexus --add-data "templates;templates" --add-data "static;static" --add-data "src;src" --add-data "scr;scr" --add-data "cdr;cdr" app.py

pyinstaller build command [Linux]

  pyinstaller --windowed --console --clean --onefile --icon=static/favicon/favicon.ico --name SearchNexus --add-data "templates:templates" --add-data "static:static" --add-data "src:src" --add-data "scr:scr" --add-data "cdr:cdr" "app.py"
